\hypertarget{md_docs_pages_notes_autotoc_md14}{}\doxysection{VIM Commands}\label{md_docs_pages_notes_autotoc_md14}

\begin{DoxyItemize}
\item Ctrl-\/w s, ctrl-\/w v -\/ Split current window half horizontally or vertically with same file in it
\item Ctrl-\/w n -\/ Same thing, new buffer but horizontal
\item \+:vnew -\/ same thing, new buffer but vertical
\item Ctrl-\/w q -\/ Close
\item Ctrl-\/w arrows -\/ move between buffers
\item \+:vs filename -\/ open file in new vertical split
